                                #include <iostream>

                                    void reverseArr(int arr[], int size)
                                    {
                                        int revArr[size];
                                        int j = 0;
                                        for (int i = size - 1; i >= 0; i--)
                                        {
                                            revArr[j] = arr[i];
                                            j++;
                                        }
                                        // printing reverse
                                        for (int j = 0; j < size; j++)
                                        {
                                            cout << "at " << j << " -> " << revArr[j] << endl;
                                        }
                                    }
                                    void revUsingSwap(int arr[], int size)
                                    {
                                        int first = 0;
                                        int last = size - 1;
                                        while (first != last)
                                        {
                                            // swaping here;
                                            int temp = arr[first];
                                            arr[first] = arr[last];
                                            arr[last] = temp;
                                            first++;
                                            last--;
                                        }
                                    }
                                    void printArr(int arr[], int size)
                                    {
                                        for (int i = 0; i < size; i++)
                                        {
                                            cout << arr[i] << " ";
                                        }
                                        cout<<endl;

1207. Unique Number of Occurrences

                            

                                //solution includes vector which is not taught yet
                                class Solution {
                                    public:
                                        bool uniqueOccurrences(vector<int>& arr) {
                                            vector<int> ans;
                                        int size = arr.size();
                                        sort(arr.begin(), arr.end());
                                        int i = 0;
                                        while (i < size)
                                        {
                                            int count = 1;
                                            for (int j = i+1; j < size; j++)
                                            {
                                                if (arr[i] == arr[j])
                                                {
                                                    count++;
                                                }
                                                else
                                                {
                                                    break;
                                                }
                                            }
                                            ans.push_back(count);
                                            i = i + count;
                                        }
                                        size = ans.size();
                                        sort(ans.begin(), ans.end());
                                        for (int i = 0; i < size - 1; i++)
                                        {
                                            if (ans[i] == ans[i + 1])
                                                return false;
                                        }
                                        return true;
                                        }
                                    };

Pair sum

                            


                                #include <bits/stdc++.h>
                                vector<vector<int>> pairSum(vector<int> &arr, int s)
                                {
                                    // Write your code here.
                                    vector<vector<int>> ans;
                                    for (int i = 0; i < arr.size(); i++)
                                    {
                                        for (int j = i + 1; j < arr.size(); j++)
                                        {
                                            if (arr[i] + arr[j] == s)
                                            {
                                                vector<int> temp;
                                                temp.push_back(min(arr[i], arr[j]));
                                                temp.push_back(max(arr[i], arr[j]));
                                                ans.push_back(temp);
                                            }
                                        }
                                    }
                                    sort(ans.begin(), ans.end());
                                    return ans;
                                }

find triplets

                            

                                /*
        
                                Time Complexity: O(N^3)
                                Space Complexity: O(1)
                            
                                Where N is the number of elements in the array.
                            
                             */
                            
                            #include
                            
                            vector<vector<int>> findTriplets(vector<int>arr, int n, int K)  {
                            
                                // Set to keep the track of visited triplets.
                                set<vector<int>>visited;
                                vector<vector<int>>ans;
                                for (int i = 0; i < n - 2; i++) {
                                    for (int j = i + 1; j < n - 1; j++) {
                                        for (int k = j + 1; k < n; k++) {
                                            // If we find a valid triplet.
                                            if (arr[i] + arr[j] + arr[k] == K) {
                                                vector<int> triplet;
                                                triplet.push_back(arr[i]);
                                                triplet.push_back(arr[j]);
                                                triplet.push_back(arr[k]);
                                                // Sorting the triplet track distinct triplets.
                                                sort(triplet.begin(), triplet.end());
                                                if (visited.find(triplet) == visited.end()) {
                                                    ans.push_back(triplet);
                                                    visited.insert(triplet);
                                                }
                                            }
                                        }
                                    }
                                }
                            
                                return ans;
                            
                            }
