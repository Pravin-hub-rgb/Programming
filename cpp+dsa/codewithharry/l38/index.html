Single Inheritance

  • Single inheritance is a type of inhertance in which a derived class is inherited with only one base class.

Public Mode

                    

                            #include <iostream>
                            using namespace std;
                            
                            class Base
                            {
                                int data1; // private by default and is not inheritable
                            public:
                                int data2;
                                void setData();
                                int getData1();
                                int getData2();
                            };
                            
                            void Base ::setData(void)
                            {
                                data1 = 10;
                                data2 = 20;
                            }
                            
                            int Base ::getData1()
                            {
                                return data1;
                            }
                            
                            int Base ::getData2()
                            {
                                return data2;
                            }
                            
                            class Derived : public Base // class is being derived publically
                            {
                                int data3;
                            
                            public:
                                void process();
                                void display();
                            };
                            
                            void Derived ::process()
                            {
                                data3 = data2 * getData1();
                            }
                            void Derived ::display()
                            {
                                cout <<<" Value of data 1 is " <<<getData1() <<<endl;
                                cout <<<" Value of data 2 is " <<<data2 <<<endl;
                                cout <<<" Value of data 3 is " <<<data3 <<<endl;
                            }
                            int main()
                            {
                                Derived der;
                                der.setData(); // This is the public function of Base class
                                der.process();
                                der.display();
                                return 0;
                            }

Private Mode

                        

                                #include <iostream>
                                using namespace std;
                                
                                class Base
                                {
                                    int data1; // private by default and is not inheritable
                                public:
                                    int data2;
                                    void setData();
                                    int getData1();
                                    int getData2();
                                };
                                
                                void Base ::setData(void)
                                {
                                    data1 = 10;
                                    data2 = 20;
                                }
                                
                                int Base ::getData1()
                                {
                                    return data1;
                                }
                                
                                int Base ::getData2()
                                {
                                    return data2;
                                }
                                
                                class Derived : private Base // class is being derived publically
                                {
                                    int data3;
                                
                                public:
                                    void process();
                                    void display();
                                };
                                
                                void Derived ::process()
                                {
                                    setData(); // we can access it over here.
                                    data3 = data2 * getData1();
                                }
                                void Derived ::display()
                                {
                                    cout << " Value of data 1 is " << getData1() << endl;
                                    cout << " Value of data 2 is " << data2 << endl;
                                    cout << " Value of data 3 is " << data3 << endl;
                                }
                                int main()
                                {
                                    Derived der;
                                    // der.setData(); // As we are inheriting in private mode so we cannot call this function
                                    // As this becomes private member of derived class.
                                    // but we can access these type of function inside another public function of derived classs.
                                    der.process();
                                    der.display();
                                    return 0;
                                }
