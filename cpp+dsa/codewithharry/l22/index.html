Nesting of Member function

  • If one member function is called inside the other member function of the same class
                

                    // class is created with string type 
                    // inputs a number and checks if it is a binary number or not 
                    // nesting of a function is done here 
                    // ones compliment is printed
                    #include <iostream>
                    #include <string> 
                        using namespace std;
                        
                        class Binary {
                            private : // by default all members of a class are private
                                string s;
                                void chk_bin(void);

                            public :
                                void read(void);
                                void ones_complement();
                        };
                        
                        void Binary :: ones_complement(){
                            chk_bin(); // Nesting of member function
                            for (int i = 0; i < s.length(); i++)
                            {
                                if(s.at(i) == '0')
                                    s.at(i) = '1';
                                else 
                                    s.at(i) = '0';
                            }
                            for (int i = 0; i < s.length(); i++)
                            {
                                cout<<s.at(i)<<" ";
                            }
                            
                            
                        }
                        
                        void Binary :: read(void){
                            cout<<"Enter the Number : ";
                            cin>>s;
                        }
                        
                        void Binary :: chk_bin(void){
                            for (int i = 0; i < s.length(); i++)
                            {
                                if(s.at(i)!= '0' && s.at(i)!= '1'){
                                    cout<<"Incorrect binary format"<<endl;
                                    exit(0);
                                }
                            }    
                        }
                        
                        int main()
                        {
                            
                            Binary bin;
                            bin.read();
                            //bin.chk_bin(); // If this function is private then we can't access it like this.
                            bin.ones_complement();
                        
                            return 0;
                        }
