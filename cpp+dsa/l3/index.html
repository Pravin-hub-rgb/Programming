Conditionals and Loops

  • Executing code according to the conditions.

If statements

                

                    if(condition){
                        //code
                    }

If the condition within () is true, then execute the entire code block within {}.

                
                    
                    if(a>b){
                        cout<< a << endl;
                    }
                    if(a<b){
                        cout<< b << endl;
                    }

In the above example, instead of checking again for b>a in the second if block, we know that if if the first if condition does not get fulfilled then the second block must be executed no matter what. This can be achieved using on if-else block. 

                
                    
                    if(a>b){
                        cout<< a << endl;
                    }
                    else{
                        cout<< b << endl;
                    }

                            using namespace std;
                            int main()
                            {
                                int a,b;
                                cout<<"Enter the value of a: ";
                                cin>>a;
                                cout<<"Enter the value of a: ";
                                cin>>b;

                                if(a>b){
                                    cout<<"a is greater then b"<<endl;
                                } 
                                if(b>a) {
                                    cout<<"b is greater then a"<<endl;
                                }
                                return 0;
                            }
  • Note: cin ignores ENTER (/n), TAB (/t) and space ( ) while taking input. These are called whitespace character use cin.get() to read these whitespace characters.

Program to tell if a number is positive, negative or zero.

                           
                           
                            #include <iostream> 
                            using namespace std;
                            int main()
                            {
                                int a;
                                cout<<"Enter a number: ";
                                cin>>a;
                                if(a>0){
                                    cout<<"Number is positive"<<endl;
                                }
                                else if(a<0){
                                    cout<<"Number is negative"<<endl;
                                } else {
                                    cout<<"Number is zero"<<endl;
                                }
                                return 0;
                            }

uppercase, lowercase or digit code

                           
                           
                            #include <iostream>
                                using namespace std;
                                int main()
                                {
                                    char ch;
                                    cout << "Enter a character : ";
                                    cin >> ch;
                                    if (ch <= 'z' && ch >= 'a')
                                    {
                                        cout << "lowercase\n";
                                    }
                                    else if (ch <= 'Z' && ch >= 'A')
                                    {
                                        cout << "uppercase\n";
                                    }
                                    else if (ch <= '9' && ch >= '0')
                                    {
                                        cout << "number\n";
                                    }
                                    else
                                    {
                                        cout << "doesn't know";
                                    }
                                    return 0;
                                }

While loop

                       
                       
                        while(...){
                            ...
                        }
  • While the condition is true, keep on executing the block.
                            
                            
                                    #include <iostream>
                                    using namespace std;
                                    int main()
                                    {
                                        int n, i = 1, sum = 0;
                                        cin >> n;
                                        while (i <= n)
                                        {
                                            sum += i;
                                            i++;
                                        }
                                        cout << "The sum is : " << sum;
                                        return 0;
                                    }
