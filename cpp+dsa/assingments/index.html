1281. Subtract the product and sum of digits of an integer

                            
class Solution {
public:
    int subtractProductAndSum(int n) {
        int sum = 0;
        int product = 1;
        int temp = n;
        while(temp>0){            
            sum+=temp%10;
            product*=temp%10;
            temp/=10;
        }
        return product - sum;
    }
};

191. Number of 1 Bits
Explaination 

                            
class Solution
{
public:
    int hammingWeight(uint32_t n)
    {
        int count = 0;
        while (n != 0)
        {
            // checking last bit
            if (n & 1)
            {
                count++;
            }
            n >>= 1;
        }
        return count;
    }
};

7. Reverse integer

                            
#include <iostream>
#include <limits.h>
using namespace std;
int reverse(int x)
{
    int answer = 0;
    while (x != 0)
    {
        int digit = x % 10;
        if ((answer > INT_MAX / 10) || (answer < INT_MIN / 10))
        {
            return 0;
        }
        answer = (answer * 10) + digit;
        x /= 10;
    }
    return answer;
}
int main()
{
    int num;
    cout << "Enter a number";
    cin >> num;
    cout << reverse(num);
    return 0;
}

1009. Complement of Base 10 Integer

                            
#include <iostream>
using namespace std;
int bitwiseComplement(int n)
{
    int m = n;
    int mask = 0;
    
    //edge case
    if (n == 0)
    {
        return 1;
    }
    while (m!=0)
    {
        mask = (mask << 1) | 1;
        m = m >> 1;
    }
    int ans = (~n) & mask;
    return ans;
}
int main()
{
    int num;
    cout << "Enter a number";
    cin >> num;
    cout << bitwiseComplement(num);
    return 0;
}

231. Power of Two

                            
#include <iostream>
#include <cmath>
using namespace std;
bool isPowerOfTwo(int n)
{
    int i = 1;
    while (pow(2, i) <= n)
    {
        if (pow(2, i) == n)
        {
            return true;
        }
        i++;
    }
    return false;
}
int main()
{
    int num;
    cout << "Enter a number : ";
    cin >> num;
    if (isPowerOfTwo(num))
    {
        cout << "True";
    }
    else
    {
        cout << "false";
    }
    return 0;
}
