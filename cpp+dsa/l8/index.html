#include <iostream>
using namespace std;
int main()
{
    int a = 2;
    switch (a)
    {
    case 1:
        cout ≪< "First" ≪< endl;
        break;
        // If case is matched then it exits out of switch
        // doesn't check further cases
    case 2:
        cout ≪< "second" ≪< endl;
        break;
    default:
    // If none of above case matches then the default runs
    // not necessary
        cout ≪< "It is a default case" ≪< endl;
    }
    return 0;
}

Nested switch ↓

                
#include <iostream>
using namespace std;
int main()
{
    char ch = 'a';
    int a = 2;
    switch (a)
    {
    case 1:
        cout << "First" << endl;
        break;
    case 2:
        cout << "Inside the second case " << endl;
        switch (ch)
        {
        case 'a':
            cout << "Printing " << ch;
            break;
        case 'b':
            cout << "Printing" << ch;
            break;

        default:
            cout << "nothing matched";
            break;
        }
        break;
    default:
        cout << "It is a default case" << endl;
    }
    return 0;
}

Calculator using switch

                            
#include <iostream>
using namespace std;
int main()
{
    int a, b;
    char op;
    cout < "Enter first number : ";
    cin >> a;
    cout < "Enter second number : ";
    cin >> b;
    cout < "Operator : ";
    cin >> op;

    switch (op)
    {
    case '+':
        cout < a < "+" < b < " = " < a + b < endl;
        break;
    case '-':
        cout < a < "-" < b < " = " < a - b < endl;
        break;
    case '*':
        cout < a < "*" < b < " = " < a * b < endl;
        break;
    case '/':
        cout < a < "/" < b < " = " < a / b < endl;
        break;
    case '%':
        cout < a < "%" < b < " = " < a % b < endl;
        break;

    default:
        cout < "can't calculate";
        break;
    }
    return 0;
}
