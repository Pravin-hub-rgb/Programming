power function

                        
#include <iostream>
using namespace std;
int power(int a, int b)
{
    int answer = 1;
    for (int i = 0; i < b; i++)
    {
        answer *= a;
    }
    return answer;
}
int main()
{
    int a, b;
    cin >> a >> b;
    cout < power(a, b);
    return 0;
}

factorial function

                            
#include <iostream>
using namespace std;

int factorial(int n)
{
    int fact = 1;
    while (n > 0)
    {
        fact *= n;
        n--;
    }
    return fact;
}

int NCR(int n, int r)
{
    int facN = factorial(n), facR = factorial(r), facNmr = factorial(n - r);
    int ncr = (facN / (facR * facNmr));
    return ncr;
}

int main()
{
    int n, r;
    cin >> n >> r;
    cout << NCR(n, r);
}
